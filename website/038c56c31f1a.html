Hexo之Just主题开发配置文件备份

运营喵 • 2020-0-5

本人第一次学习开发Hexo主题，期间遇到过很多坑，还好百度是最好的老师，本文记录或为以后再次开发主题之用，也希望可以帮助一些和我同样第一次接触的同学。特此备份记录。

# 本主题使用到插件（需手动执行安装）

# 外链自动加NOFOLLOW
# npm install hexo-autonofollow --save
# npm uninstall hexo-autonofollow
nofollow:
  enable: true
  exclude:
    - www.lopins.cn
    - www.chuying.ltd
    - www.docnn.cn
    - www.15234.net

# 生成百度SiteMap索引并提交搜索引擎
# npm install hexo-generator-baidu-sitemap --save
# npm uninstall hexo-generator-baidu-sitemap
# 生成谷歌SiteMap索引并提交搜索引擎
# npm install hexo-generator-sitemap --save
# npm uninstall hexo-generator-sitemap
#地图配置：不需要的页面可在页面中添加
#---
#baidusitemap: false
#sitemap: false
#---

sitemap:
  path: sitemap.xml
  #template: ./sitemap_template.xml
  rel: true
  tags: true
  categories: true

baidusitemap:
  path: baidusitemap.xml
  tags: true
  categories: true
